// Import
import { Uploader } from '@ldesign/upload'

// Create instance
const uploader = new Uploader({
  container: '#uploader',
  endpoint: '/api/upload',
  autoUpload: false,
  validation: {
    accept: 'image/*,video/*',
    maxSize: 10 * 1024 * 1024 // 10MB
  }
})

// Listen to events
uploader.on('fileAdded', (file) => {
  console.log('File added:', file.file.name)
})

uploader.on('uploadSuccess', (fileId, response) => {
  console.log('Upload successful:', response)
})

// Methods
uploader.uploadAll()       // Upload all files
uploader.clear()           // Clear all files
uploader.openFilePicker()  // Open file picker
